协作过滤

协作过滤(Collaborative Filtering)算法：对一大群人经行搜索，从中找出与和我们品味相近的一小群人。算法对这一小群人偏爱的其他内容进行考查，并将其组合起来构造一个进过排序的推荐列表。

算法:

  • 欧几里得距离(Euclidean Distance Score) ：以物品数量n为维度，计算用户在n维空间中的坐标点之间欧式距离
  • 皮尔逊相关度(Pearson Correlation Score) ：以两个用户为维度(2维)，计算出物品坐标和最佳拟合直线的相关程度。
  • Jaccard 系数
  • 曼哈顿距离
    推荐方式：
  • 基于用户过滤
  • 基于商品过滤

def sim_pearson(pd_set,person1,person2):
    if len((pd_set.loc[person1]+pd_set.loc[person2]).dropna())==0:
        return 0
    person1_clean=(pd_set.loc[person1]+pd_set.loc[person2]0).dropna()
    person2_clean=(pd_set.loc[person1]0+pd_set.loc[person2]).dropna()
    person1_avg=sum(person1_clean)/len(person1_clean)
    person2_avg=sum(person2_clean)/len(person2_clean)
    cor1=sum((person1_clean-person1_avg).T*(person2_clean-person2_avg).T)
    cor2=sqrt(sum((person1_clean-person1_avg).T2))sqrt(sum((person2_clean-person2_avg).T**2))
    if cor2==0:
        return 0
    return cor1/cor2

def topMatches(pd_set,person,n=5,similarity=sim_pearson):
    scores=[(similarity(pd_set,person,other),other)
            for other in pd_set.T if other!=person]
    scores.sort()
    scores.reverse()
    return scores[0:n]

def getRecommendations(pd_set,person,similarity1=sim_pearson):
    pd_temp=pd_set.drop(labels=person)
    pd_temp1=pd_set.drop(labels=person)
    for other in pd_temp.T:
        pd_temp.loc[other]=similarity1(pd_set,person,other)
        pd_temp1.loc[other]=pd_temp1.loc[other]*0+similarity1(pd_set,person,other)
    rankings=[(sum(pd_temp[movie].dropna())/sum(pd_temp1[movie].dropna()),movie)
               for movie in pd_temp if movie not in pd_set.T[person].dropna() ]
    rankings.sort()
    #rankings.reserve()
    return rankings
